## 문제

[https://www.acmicpc.net/problem/11005](https://www.acmicpc.net/problem/11005)

## 코드

```c++
#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
    int n; //10진법
    int b; //바꿀 진법

    string answer="";
    cin>>n>>b;

    while(n>0){
        int x;//나머지
        x=n%b;
        if(x>=10){
            answer+=(char)(x-10+'A');
        }else{
            answer+=to_string(x);
        }
        n=n/b;
    }
    reverse(answer.begin(),answer.end());
    cout << answer;
    return 0;
}

```
