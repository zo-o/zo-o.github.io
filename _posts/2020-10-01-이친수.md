```python
n=int(input())

d=[0]*(n+1)

def dp(n):
  if d[n]>0:
    return d[n]
  elif n==1:
    d[1]=1
  elif n==2:
    d[2]=1
  else:
    d[n]=dp(n-1)+dp(n-2)
  return d[n]

print(dp(n))
```
